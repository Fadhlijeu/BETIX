    function safeParseUser(raw) {
      if (!raw) return null;
      try { return JSON.parse(raw); } catch { return null; }
    }

    function getInitials(name, email) {
      var source = '';
      if (typeof name === 'string' && name.trim()) {
        source = name.trim();
      } else if (typeof email === 'string' && email.trim()) {
        source = email.split('@')[0].replaceAll('.', ' ').replaceAll('_', ' ').replaceAll('-', ' ').trim();
      }
      if (!source) return '?';
      var parts = source.split(' ').filter(Boolean);
      var first = parts[0] ? parts[0].charAt(0) : '';
      var second = parts.length > 1 ? parts[1].charAt(0) : (source.length > 1 ? source.charAt(1) : '');
      return (first + second).toUpperCase() || '?';
    }

    function syncUserProfile() {
      var user = null;
      try { user = safeParseUser(localStorage.getItem('user_session')); } catch { }
      var name = 'Pengguna';
      var email = '';
      if (user && typeof user === 'object') {
        name = user.name || user.nama || user.username || name;
        email = user.email || email;
      }
      var profileName = document.querySelector('#page-profil .profile-name');
      var profileEmail = document.querySelector('#page-profil .profile-email');
      var avatarBig = document.querySelector('#page-profil .avatar-big');
      var headerProfileBtn = document.querySelector('.header-actions .profile-btn');
      if (profileName) profileName.textContent = name;
      if (profileEmail) profileEmail.textContent = email;
      var initials = getInitials(name, email);
      if (avatarBig) avatarBig.textContent = initials;
      if (headerProfileBtn) headerProfileBtn.textContent = initials;

      // Update profile stats from stored tickets
      try {
        var tickets = JSON.parse(localStorage.getItem('betix_active_tickets') || '[]');
        var statTicket = document.querySelector('#page-profil .stat-val:not(.gold):not(.red)');
        if (statTicket) statTicket.textContent = tickets.length;
        if (user && user.points) {
          var statPoin = document.querySelector('#page-profil .stat-val.gold');
          if (statPoin) statPoin.textContent = user.points;
        }
      } catch { }
    }

    function updatePaymentSummary() {
      try {
        var state = window.BE_TIX && window.BE_TIX.state;
        if (!state) return;
        var filmEl = document.getElementById('pay-film');
        var cinemaEl = document.getElementById('pay-cinema');
        var seatsEl = document.getElementById('pay-seats');
        var priceEl = document.getElementById('pay-price');
        var totalEl = document.getElementById('pay-total');
        if (filmEl && state.selectedMovie) filmEl.textContent = state.selectedMovie.title || '-';
        if (cinemaEl && state.selectedCinema) cinemaEl.textContent = state.selectedCinema.name || '-';
        if (seatsEl) seatsEl.textContent = (state.selectedSeats || []).join(', ') || '-';
        var showtime = state.selectedShowtime;
        var price = showtime ? Number(showtime.price || 50000) : 50000;
        var count = (state.selectedSeats || []).length;
        if (priceEl) priceEl.textContent = 'Rp ' + price.toLocaleString('id-ID') + ' x ' + count;
        if (totalEl) totalEl.textContent = 'Rp ' + ((price * count) + 3000).toLocaleString('id-ID');
      } catch { }

  const SESSION_KEY = 'user_session';
  const loginForm = document.getElementById('login-form');
  const errorBox = document.getElementById('error-box');

  function safeParse(raw) {
    try {
      return raw ? JSON.parse(raw) : null;
    } catch {
      return null;
    }
  }

  function normalizeUser(user) {
    if (!user || typeof user !== 'object') return null;
    return {
      ...user,
      role: String(user.role || 'user').toLowerCase(),
      name: user.name || user.nama || user.username || '',
      email: user.email || ''
    };
  }

  function redirectByRole(user) {
    const role = String(user?.role || 'user').toLowerCase();
    if (role === 'admin') {
      window.location.assign('/admin.html');
      return;
    }
    window.location.assign('/index.html');
  }

  const existing = normalizeUser(safeParse(localStorage.getItem(SESSION_KEY)));
  if (existing) {
    redirectByRole(existing);
  }

  loginForm.addEventListener('submit', async (e) => {
    e.preventDefault();
    errorBox.style.display = 'none';

    const email = document.getElementById('email').value.trim();
    const password = document.getElementById('password').value;

    try {
      const response = await fetch('/login', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, password })
      });

      const data = await response.json();

      if (!response.ok || !data.success) {
        errorBox.textContent = data.message || 'Login gagal. Silakan coba lagi.';
        errorBox.style.display = 'block';
        return;
      }

      const user = normalizeUser(data.user);
      localStorage.setItem(SESSION_KEY, JSON.stringify(user));
      redirectByRole(user);
    } catch (err) {
      errorBox.textContent = 'Terjadi kesalahan koneksi ke server.';
      errorBox.style.display = 'block';
    }
  });

<script>
  const SESSION_KEY = 'user_session';
  const regForm = document.getElementById('register-form');
  const errorBox = document.getElementById('error-box');
  const successBox = document.getElementById('success-box');

  function clearMessages() {
    errorBox.style.display = 'none';
    successBox.style.display = 'none';
    errorBox.textContent = '';
    successBox.textContent = '';
  }

  function setRole(role, btn) {
    document.getElementById('role').value = role;

    document.querySelectorAll('.role-btn').forEach((b) => b.classList.remove('active'));
    btn.classList.add('active');

    const adminField = document.getElementById('admin-pass-field');
    const adminInput = document.getElementById('admin_passcode');

    if (role === 'admin') {
      adminField.style.display = 'block';
      adminInput.required = true;
    } else {
      adminField.style.display = 'none';
      adminInput.required = false;
      adminInput.value = '';
    }
  }

  function normalizeUser(user) {
    if (!user || typeof user !== 'object') return null;
    return {
      ...user,
      role: String(user.role || 'user').toLowerCase(),
      name: user.name || user.nama || user.username || '',
      email: user.email || ''
    };
  }

  function redirectByRole(user) {
    const role = String(user?.role || 'user').toLowerCase();
    if (role === 'admin') {
      window.location.assign('/admin.html');
      return;
    }
    window.location.assign('/index.html');
  }

  const existing = normalizeUser((() => {
    try { return JSON.parse(localStorage.getItem(SESSION_KEY)); } catch { return null; }
  })());
  if (existing) redirectByRole(existing);

  regForm.addEventListener('submit', async (e) => {
    e.preventDefault();
    clearMessages();

    const name = document.getElementById('name').value.trim();
    const email = document.getElementById('email').value.trim();
    const password = document.getElementById('password').value;
    const confirm = document.getElementById('confirm_password').value;
    const role = document.getElementById('role').value;
    const adminPasscode = document.getElementById('admin_passcode').value.trim();

    if (password !== confirm) {
      errorBox.textContent = 'Konfirmasi password tidak cocok.';
      errorBox.style.display = 'block';
      return;
    }

    const payload = { name, email, password, role };
    if (role === 'admin') payload.adminPasscode = adminPasscode;

    try {
      const response = await fetch('/register', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });

      const data = await response.json();

      if (!response.ok || !data.success) {
        errorBox.textContent = data.message || 'Registrasi gagal.';
        errorBox.style.display = 'block';
        return;
      }

      const user = normalizeUser(data.user || { name, email, role });
      localStorage.setItem(SESSION_KEY, JSON.stringify(user));
      successBox.textContent = 'Registrasi berhasil. Mengalihkan halaman...';
      successBox.style.display = 'block';

      setTimeout(() => redirectByRole(user), 800);
    } catch (err) {
      errorBox.textContent = 'Terjadi kesalahan sistem.';
      errorBox.style.display = 'block';
    }
  });
</script>
